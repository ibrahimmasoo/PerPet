If the admin page keeps saying "Checking admin access..." or loads with limited data,
publish these Firestore rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isSignedIn() {
      return request.auth != null;
    }

    function isAdmin() {
      return request.auth != null
        && request.auth.token.email == "ibrahimashrf301@gmail.com";
    }

    match /contactMessages/{docId} {
      allow create: if true;
      allow read, update, delete: if isAdmin();
    }

    match /siteContent/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /pets/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }

    match /analytics/{docId} {
      allow read, write: if isAdmin();
    }

    match /users/{userId} {
      allow read, write: if isAdmin() || (isSignedIn() && request.auth.uid == userId);
    }
  }
}
